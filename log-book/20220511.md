##### 453 最小操作使元素相等

**python解法**

```python
from typing import List

class Solution(object):
    def minMoves(self, nums:List[int])->int:
        """
        对函数的方法进行说明
        :type nums: List[int]
        :rtype: int
        """
        min_num = min(nums)
        res = 0
        for num in nums:
            res += num - min_num
        return res

nums = [1,2,4]
s = Solution()
res = s.minMoves(nums)
print(f"结果为：{res}")
```

**c++解法**

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
//1.使用数学归纳法
//长度为1时，直接返回0
//长度为2时，返回Max-Min
//长度为3时，返回Max-Min+1
//长度为4时，[1,2,2,2]结果为[4,4,4,4],次数为3；[1,1,2,2]结果为[3,3,3,3],次数为2
//[1,2,3,4]与[1,3,3,3]结果为[7,7,7,7],次数为6,满足Max-Min+3.可见，数组长度变化，
//归纳的公式就会不同，不采用
//2.n个数，每次操作n-1个数加1，加1的数为非最大的数，直到将所有的数加到相同；反过来，
//每次将最大的一个数减1，最终减成最小的数，如[1,2,4]->[1,2,3]->[1,2,2]->[1,1,2]->[1,1,1]
//计算时，直接(4-1)+(2-1)=4
class Solution;

void test01();

class Solution
{
public:
    int minMoves(vector<int>& nums)
    {
        //min_element(pos1,pos2)，查找容器指定范围中的最小值，参数为两个迭代器的位置
        //返回值为一个迭代器，可以使用*解出内容
        //cout << *min_element(nums.begin(), nums.end()) << endl;
        int min_num = *min_element(nums.begin(), nums.end());
        int res = 0;

        for (int i = 0; i < (signed)nums.size(); i++)
        {
            res = res + nums[i] - min_num;
        }
        //新式循环方法
//        for (auto num : nums)
//        {
//            res += num - min_num;
//        }
        return res;
    }
};

void test01()
{
    vector<int> nums = {1,2,4};
    Solution s;
    cout << s.minMoves(nums) << endl;
}

{
    test01();
    cout << "Hello world!" << endl;
    return 0;
}
```

##### 942 数组的增减匹配

**python解法**

```python
from typing import List

class Solution:
    def diStringMatch(self, s: str) -> List[int]:
        lens = len(s)
        #print("字符串长度为：%d" %len(s))
        print("字符串长度为：%d" %lens)

        pointer_b = 0
        pointer_e = lens
        res = []
        
        for ch in s:
            if ch == 'I':
                res.append(pointer_b)
                pointer_b = pointer_b + 1
            else:
                res.append(pointer_e)
                pointer_e = pointer_e - 1

        res.append(pointer_b)
        return res


s = "IDID"
solution = Solution()
res = solution.diStringMatch(s)
print(f"结果为：{res}")
```

**c++解法**

```cpp
#include <iostream>
#include <string>
#include <vector>

using namespace std;
//贪心算法与双指针，如果第一个字符为I，说明第一个数字要小于第二个数字，
//那么就取最小的数字，pointer[0]
//如果第一个字符为I，说明第一个数字要小于第二个数字，那么就取最大的数字，pointer[n]
//然后使用双指针重复此操作
class Solution;

void test01();
void print_vector(const vector<int>& v);

class Solution
{
public:
    vector<int> diStringMatch(string s)
    {
        int len = s.length();
        int pointer_b = 0;
        int pointer_e = len;
        vector<int> res;

        for (int i = 0; i < len; i++)
        {
            if (s[i] == 'I')
            {
                res.push_back(pointer_b);
                pointer_b++;
            }
            else
            {
                res.push_back(pointer_e);
                pointer_e--;
            }
        }
        //数组的长度要比字符串的长度大一位，最终两个指针会指向同一个值，将其补上
        res.push_back(pointer_b);

        return res;
    }
};

void test01()
{
    string s = "IDID";
    Solution solution;
    vector<int> res = solution.diStringMatch(s);
    print_vector(res);
}

void print_vector(const vector<int>& v)
{
    for (vector<int>::const_iterator it = v.begin(); it != v.end(); it++)
    {
        cout << *it << " ";
    }
    cout << endl;
}

int main()
{
    test01();
    cout << "Hello world!" << endl;
    return 0;
}
```
